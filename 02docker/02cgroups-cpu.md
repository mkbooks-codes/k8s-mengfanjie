```
cd /sys/fs/cgroup/cpu/cpudemo

# 把进程添加到 cgroup 进程配置组
# echo ps -ef|grep busyloop|grep -v grep|awk '{print $2}' > cgroup.procs

# 设置 cpuquota

cat cpu.cfs_quota_us
```
